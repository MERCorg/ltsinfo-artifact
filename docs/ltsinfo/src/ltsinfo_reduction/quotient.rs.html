8/// A trait for partition refinement algorithms that expose the block number for
9/// every state. Can be used to compute the quotient labelled transition system.
10///
11/// The invariants are that the union of all blocks is the original set, and
12/// that each block contains distinct elements
13pub trait Partition {
14    /// Returns the block number for the given state.
15    fn block_number(&self, state_index: usize) -> usize;
16
17    /// Returns the number of blocks in the partition.
18    fn num_of_blocks(&self) -> usize;
19
20    /// Returns the number of elements in the partition.
21    fn len(&self) -> usize;
22
23    /// Returns whether the partition is empty.
24    fn is_empty(&self) -> bool {
25        self.len() == 0
26    }
27
28    /// Returns true iff the partitions are equal, runs in O(n^2)
29    fn equal(&self, other: &impl Partition) -> bool {
30        // Check that states in the same block, have a single (unique) number in
31        // the other partition.
32        for block_index in 0..self.num_of_blocks() {
33            let mut other_block_index = None;
34
35            for state_index in (0..self.len()).filter(|&state_index| self.block_number(state_index) == block_index) {
36                match other_block_index {
37                    None => other_block_index = Some(other.block_number(state_index)),
38                    Some(other_block_index) => {
39                        if other.block_number(state_index) != other_block_index {
40                            return false;
41                        }
42                    }
43                }
44            }
45        }
46
47        for block_index in 0..other.num_of_blocks() {
48            let mut other_block_index = None;
49
50            for state_index in (0..self.len()).filter(|&state_index| other.block_number(state_index) == block_index) {
51                match other_block_index {
52                    None => other_block_index = Some(self.block_number(state_index)),
53                    Some(other_block_index) => {
54                        if self.block_number(state_index) != other_block_index {
55                            return false;
56                        }
57                    }
58                }
59            }
60        }
61
62        true
63    }
64}

66/// Returns a new LTS based on the given partition.
67///
68/// All states in a single block are replaced by a single representative state.
69pub fn quotient_lts(
70    lts: &LabelledTransitionSystem,
71    partition: &impl Partition,
72    eliminate_tau_loops: bool,
73) -> LabelledTransitionSystem {
74    let start = std::time::Instant::now();
75    // Introduce the transitions based on the block numbers
76    let mut transitions: Vec<(usize, CompactTransition)> = Vec::default();
77
78    for state_index in lts.iter_states() {
79        for (label, to) in lts.outgoing_transitions(state_index) {
80            let block = partition.block_number(state_index);
81            let to_block = partition.block_number(to);
82
83            // If we eliminate tau loops then check if the 'to' and 'from' end up in the same block
84            if !(eliminate_tau_loops && lts.is_hidden_label(label) && block == to_block)
85            {
86                debug_assert!(
87                    partition.block_number(state_index) < partition.num_of_blocks(),
88                    "Quotienting assumes that the block numbers do not exceed the number of blocks"
89                );
90
91                // Make sure to keep the outgoing transitions sorted.
92                transitions.push((block, CompactTransition::new(label, to_block)));
93            }
94        }
95    }
96
97    // Remove duplicates.
98    transitions.sort_unstable();
99    transitions.dedup();
100
101    let result = LabelledTransitionSystem::new(
102        partition.block_number(lts.initial_state_index()),
103        Some(partition.num_of_blocks()),
104        || transitions.iter().cloned(),
105        lts.labels().into(),
106        lts.hidden_labels().into()
107    );
108    debug!("Time quotient: {:.3}s", start.elapsed().as_secs_f64());
109    result
110}

112// Implementation of the quotient function for branching bisimulation.
113pub fn quotient_lts_efficient(
114    lts: &LabelledTransitionSystem,
115    partition: &BlockPartition
116) -> LabelledTransitionSystem {
117    let start = std::time::Instant::now();
118    let mut transitions: Vec<(usize, CompactTransition)> = Vec::default();
119
120    for block in 0..partition.num_of_blocks() {
121            // Check if state is bottom
122        let mut candidate = if let Some(state) = partition.iter_block(block).next() {
123                state
124            } else {
125                panic!("Found empty block {}", block);
126            };
127        let mut found = false;
128        // DFS into a bottom state.
129        while !found {
130            found = true;
131            for trans in lts.outgoing_transitions_compact(candidate) {
132                if lts.is_hidden_label(trans.label()) && partition.block_number(trans.state()) == block {
133                    found = false;
134                    candidate = trans.state();
135                    break;
136                } 
137            }    
138        }
139        for trans in lts.outgoing_transitions_compact(candidate) {
140            //candidate is a bottom state, so add all transitions.
141            if (trans.label() == 0) && (partition.block_number(trans.state()) == block) {
142                log::error!("Warning: this state is not bottom {}", block);
143            }
144            transitions.push((block, CompactTransition::new(trans.label(), partition.block_number(trans.state()))));
145        }
146        debug_assert!(!partition.block(block).is_empty(), "Blocks in the partition should not be empty");
147    }
148    // Remove duplicates.
149    transitions.sort_unstable();
150    transitions.dedup();
151
152    let result = LabelledTransitionSystem::new(
153        partition.block_number(lts.initial_state_index()),
154        Some(partition.num_of_blocks()),
155        || transitions.iter().cloned(),
156        lts.labels().into(),
157        lts.hidden_labels().into()
158    );
159    debug!("Time quotient: {:.3}s", start.elapsed().as_secs_f64());
160    result
161}
