20#[derive(Error, Debug)]
21pub enum IOError {
22    #[error("Invalid .aut header {0}")]
23    InvalidHeader(&'static str),
24
25    #[error("Invalid transition line")]
26    InvalidTransition(),
27}

29///     `(<from>: Nat, "<label>": Str, <to>: Nat)`
30///     `(<from>: Nat, <label>: Str, <to>: Nat)`
31fn read_transition(input: &str) -> Result<(&str, &str, &str), Box<dyn Error>> {
32    let start_paren = input.find('(').ok_or(IOError::InvalidTransition())?;
33    let start_comma = input.find(',').ok_or(IOError::InvalidTransition())?;
34
35    // Find the comma in the second part
36    let start_second_comma = input.rfind(',').ok_or(IOError::InvalidTransition())?;
37    let end_paren = input.rfind(')').ok_or(IOError::InvalidTransition())?;
38
39    let from = &input[start_paren+1..start_comma].trim();
40    let label = &input[start_comma+1..start_second_comma].trim();
41    let to = &input[start_second_comma+1..end_paren].trim();
42    // Handle the special case where it has quotes.
43    if label.starts_with('"') && label.ends_with('"') {
44        return Ok((from, &label[1..label.len()-1], to))
45    }
46
47    Ok((from, label, to))
48}

50/// Loads a labelled transition system in the Aldebaran format from the given reader.
51///
52/// The Aldebaran format consists of a header:
53///     `des (<initial>: Nat, <num_of_transitions>: Nat, <num_of_states>: Nat)`
54///     
55/// And one line for every transition:
56///     `(<from>: Nat, "<label>": Str, <to>: Nat)`
57///     `(<from>: Nat, <label>: Str, <to>: Nat)`
58pub fn read_aut(reader: impl Read, mut hidden_labels: Vec<String>) -> Result<LabelledTransitionSystem, Box<dyn Error>> {
59    let start = Instant::now();
60    debug!("Reading LTS in .aut format...");
61
62    let mut lines = LineIterator::new(reader);
63    lines.advance();
64    let header = lines
65        .get()
66        .ok_or(IOError::InvalidHeader("The first line should be the header"))?;
67
68    // Regex for des (<initial>: Nat, <num_of_states>: Nat, <num_of_transitions>: Nat)
69    let header_regex = Regex::new(r#"des\s*\(\s*([0-9]*)\s*,\s*([0-9]*)\s*,\s*([0-9]*)\s*\)\s*"#)
70        .expect("Regex compilation should not fail");
71
72    let (_, [initial_txt, num_of_transitions_txt, num_of_states_txt]) = header_regex
73        .captures(header)
74        .ok_or(IOError::InvalidHeader(
75            "does not match des (<init>, <num_of_transitions>, <num_of_states>)",
76        ))?
77        .extract();
78
79    let initial_state: usize = initial_txt.parse()?;
80    let num_of_transitions: usize = num_of_transitions_txt.parse()?;
81    let num_of_states: usize = num_of_states_txt.parse()?;
82
83    // This is used to keep track of the label to index mapping.
84    let mut labels_index: HashMap<String, LabelIndex> = HashMap::new();
85    let mut labels: Vec<String> = Vec::new();
86
87    // Pre-allocate the transitions vector
88    let mut transitions: Vec<(usize, CompactTransition)> = Vec::with_capacity(num_of_transitions);
89    let mut progress = Progress::new(
90        |value, increment| debug!("Reading transitions {}%...", value / increment),
91        num_of_transitions,
92    );
93
94    while let Some(line) = lines.next() {
95        trace!("{}", line);
96        let (from_txt, label_txt, to_txt) = read_transition(line)?;
97
98        // Parse the from and to states, with the given label.
99        let from: usize = from_txt.parse()?;
100        let to: usize = to_txt.parse()?;
101
102        let label_index = *labels_index.entry(label_txt.to_string()).or_insert(labels.len());
103
104        if label_index >= labels.len() {
105            labels.resize_with(label_index + 1, Default::default);
106        }
107
108        trace!("Read transition {} --[{}]-> {}", from, label_txt, to);
109
110        // Create CompactTransition directly
111        transitions.push((from, CompactTransition::new(label_index, to)));
112
113        if labels[label_index].is_empty() {
114            labels[label_index] = label_txt.to_string();
115        }
116
117        progress.add(1);
118    }
119
120    // Remove duplicated transitions, it is not clear if they are allowed in the .aut format.
121    // I don't think we should do this, it seems expensive and not really necessary.
122    // transitions.sort_unstable_by_key(|(from, _)| *from);
123    // transitions.dedup();
124
125    debug!("Finished reading LTS");
126
127    hidden_labels.push("tau".to_string());
128    debug!("Time read_aut: {:.3}s", start.elapsed().as_secs_f64());
129
130    Ok(LabelledTransitionSystem::new(
131        initial_state,
132        Some(num_of_states),
133        || transitions.iter().cloned(),
134        labels,
135        hidden_labels,
136    ))
137}

139/// Write a labelled transition system in plain text in Aldebaran format to the given writer.
140pub fn write_aut(writer: &mut impl Write, lts: &LabelledTransitionSystem) -> Result<(), Box<dyn Error>> {
141    writeln!(
142        writer,
143        "des ({}, {}, {})",
144        lts.initial_state_index(),
145        lts.num_of_transitions(),
146        lts.num_of_states()
147    )?;
148
149    for state_index in lts.iter_states() {
150        for trans in lts.outgoing_transitions_compact(state_index) {
151            writeln!(
152                writer,
153                "({}, \"{}\", {})",
154                state_index,
155                if lts.is_hidden_label(trans.label()) {
156                    "tau"
157                } else {
158                    &lts.labels()[trans.label()]
159                },
160                trans.state()
161            )?;
162        }
163    }
164
165    Ok(())
166}
